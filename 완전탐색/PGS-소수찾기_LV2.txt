import java.util.*;
import java.math.*;
class Solution {
    
    public static ArrayList<String> numberArr;
    static boolean[] visited;
    static String tmp;
    
    public boolean isPrime(double x){
        double end = Math.sqrt(x);
        if(x==1) return false;
        for(int i=1;i<end;i++){
            if(x%i==0) return false;
        }
        return true;
    }
    
    public void makeNum(String numbers, int n, int d){
        //n : 뽑을 개수
        //d : 깊이
        if(d==n) {
            numberArr.add(tmp);
            tmp = "";
            return;
        }
        else{
            for(int i=0;i<n;i++){
                if(!visited[i]){
                    visited[i] = true;
                    tmp += numbers.charAt(i);
                    makeNum(numbers, n, d+1);
                    visited[i] = false;
                }
            }
        }
        
    }
    
    
    public int solution(String numbers) {
        visited = new boolean[numbers.length()];
        numberArr = new ArrayList<>();
        
        for(int i=1;i<=numbers.length();i++){
            makeNum(numbers,i,0);
        }
        
        for(String s:numberArr){
            System.out.println(s);
        }
        
        return 0;
        
        
    }
}
